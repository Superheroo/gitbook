swiper插件

swiper官网：swiper

在页面中引入所需文件：

<script src="js/jquery-2.1.0.js" type="text/javascript" charset="utf-8"></script>
<link rel="stylesheet" type="text/css" href="css/swiper-3.4.2.min.css"/>
<script src="js/swiper-3.4.2.min.js" type="text/javascript" charset="utf-8"></script>

swiper插件之轮播图

<!DOCTYPE html>
<html>
    <head>
        <script src="js/jquery-2.1.0.js" type="text/javascript" charset="utf-8"></script>
        <link rel="stylesheet" type="text/css" href="css/swiper-3.4.2.min.css"/>
        <script src="js/swiper-3.4.2.min.js" type="text/javascript" charset="utf-8"></script>
        <meta charset="UTF-8">
        <title></title>
        <style>
            .lunbo{
                position: relative;
                width: 540px;
                height: 300px;
                overflow: hidden;
                margin: 0 auto;
            }
            .all_tu{
                width: 999999px;

            }
            .lunbo_tu{
                float: left;
                width: 540px;
                height: 300px;
                overflow: hidden;
            }
            .lunbo_tu img{
                width: 100%;
            }
            .btn{
                font-size: 50px;
                position:absolute;
                top:50%;
                transform: translateY(-50%);
                z-index: 999;
                cursor: pointer;
                color: white;
            }
            .btn:nth-child(3){
                left: 10px;
            }
            .btn:nth-child(4){
                right: 10px;
            }
        </style>
    </head>
    <body>
    <div class="lunbo swiper-container">
        <!--图片区-->
        <div class="all_tu swiper-wrapper">
            <div class="lunbo_tu swiper-slide"><img src="img/1big.jpg"></div>
            <div class="lunbo_tu swiper-slide"><img src="img/2big.jpg"></div>
            <div class="lunbo_tu swiper-slide"><img src="img/3big.jpg"></div>
            <div class="lunbo_tu swiper-slide"><img src="img/5big.jpg"></div>
        </div>
        <!--小圆点控制器-->
        <div class="swiper-pagination"></div>
        <!--左右按钮-->
        <div class="btn swiper-button-prev">&lt;</div>
        <div class="btn swiper-button-next">&gt;</div>

    </div>
        <script>
            $(function(){
                var lunbotu=new Swiper(".lunbo",{
                    autoplay:2000,/*轮播时间*/
                    loop:true,/*实现无限轮播*/
                    pagination:'.swiper-pagination',/*小圆点*/
                    prevButton:'.swiper-button-prev',
                    nextButton:'.swiper-button-next'
                });
            })
        </script>
    </body>
